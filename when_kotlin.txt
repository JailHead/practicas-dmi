fun main() {
    println("=== DETERMINADOR DE LUGAR DE TRABAJO ===")
    println("Ingrese una profesión para conocer su lugar de trabajo:")
    println("Profesiones disponibles:")
    println("- Hospital: Doctor, Doctora, Enfermero, Enfermera")
    println("- Oficina: Director, Directora, Secretario, Secretaria")
    println("- Iglesia: Monja, Monje, Feligres, Feligresa")
    println()
    
    print("Ingrese la profesión: ")
    val profesion = readLine()?.trim()?.lowercase()
    
    if (profesion.isNullOrEmpty()) {
        println("Error: No se ingresó ninguna profesión.")
        return
    }
    
    val lugarTrabajo = when (profesion) {
        "doctor", "doctora", "enfermero", "enfermera" -> "Hospital"
        "director", "directora", "secretario", "secretaria" -> "Oficina"
        "monja", "monje", "feligres", "feligresa" -> "Iglesia"
        else -> "Profesión no reconocida"
    }
    
    println()
    if (lugarTrabajo != "Profesión no reconocida") {
        println("✓ La profesión '$profesion' trabaja en: $lugarTrabajo")
    } else {
        println("✗ La profesión '$profesion' no está en nuestra base de datos.")
        println("Por favor, verifique que haya escrito correctamente el nombre de la profesión.")
    }

}
